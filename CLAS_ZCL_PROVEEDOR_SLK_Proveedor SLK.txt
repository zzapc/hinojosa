  DATA l_lifnr TYPE lifnr.

  SELECT SINGLE lifn2

    FROM wyt3 JOIN t024e ON wyt3~ekorg = t024e~ekorg
   WHERE lifnr = reguh-lifnr
*     and parvw = 'RS'   "Destinatario de pago
     and parvw = 'VA'   "Dirección pagaré
     AND bukrs = reguh-zbukr.

  IF sy-subrc = 0 AND l_lifnr NE reguh-lifnr AND  NOT L_LIFNR IS INITIAL.
    SELECT SINGLE land1 name1 stras regio

METHOD get_desc_form_pago.

  CLEAR text.
  SELECT SINGLE text1 FROM  t052u
    INTO text
   WHERE spras  = spras
     AND zterm  = zterm
     AND ( ztagg = 00 OR ztagg = 10 ). "?????
  IF sy-subrc NE 0.
    CLEAR text.
    SELECT SINGLE text1 FROM  t052u
      INTO text
     WHERE spras  = spras
       AND zterm  = zterm.
    IF sy-subrc NE 0.
      SELECT SINGLE text1 FROM  t052u
        INTO text
       WHERE spras  = spras
         AND zterm  = 'E'.
      IF sy-subrc NE 0.
        SELECT SINGLE text1 FROM  t052u
          INTO text
         WHERE spras  = spras.
      ENDIF.
    ENDIF.
  ENDIF.

ENDMETHOD.
METHOD GET_DESC_VIA_PAGO.

  CLEAR text.
  SELECT SINGLE text2 FROM  t042zt
    into text
   WHERE spras  = spras
     AND land1  = land1
     AND zlsch  = zlsch.

  SELECT SINGLE text2 FROM  t042zt
    into text
   WHERE spras  = spras
     AND land1  = 'ES'
     AND zlsch  = zlsch.
  IF sy-subrc NE 0.
    SELECT SINGLE text1 FROM  t042z
      into text
     WHERE land1  = land1
       AND zlsch  = zlsch.
  ENDIF.

ENDMETHOD.
