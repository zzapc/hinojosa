*&---------------------------------------------------------------------*
*&      Module  USER_COMMAND_0001  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE user_command_0001 INPUT.
  CASE sy-ucomm.
    WHEN 'ENTE'.
      MESSAGE 'Pulse grabar para registrar el consumo' TYPE 'I'.

    WHEN 'SAVE'.
      IF zwm_intf_bhs-metros_restantes EQ space AND sy-binpt EQ space.
        DATA l_answer(1).

        CALL FUNCTION 'POPUP_TO_CONFIRM'
          EXPORTING
            text_question  = 'Está grabando un consumo total ¿Desea continuar?'
            text_button_1  = 'Sí'
            text_button_2  = 'No'
          IMPORTING
            answer         = l_answer
          EXCEPTIONS
            text_not_found = 1
            OTHERS         = 2.

        CHECK l_answer EQ '1'.
      ENDIF.


      PERFORM f_consumo_manual USING ltbk-nlpla
                                     ltbk-benum
                                     lqua
                                     pedido1
                                     pedido2
                                     zwm_intf_bhs-metros_restantes
                                     space.


      MESSAGE 'Consumo registrado con éxito' TYPE 'S'.

      LEAVE PROGRAM.

    WHEN 'BACK' OR 'CANCEL'.
      LEAVE PROGRAM.

  ENDCASE.
ENDMODULE.
*&---------------------------------------------------------------------*
*&      Module  EXIT  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE exit INPUT.
  LEAVE PROGRAM.
ENDMODULE.
*&---------------------------------------------------------------------*
*&      Module  VALIDAR_UA  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE validar_ua INPUT.
  SELECT SINGLE * FROM lqua
    WHERE lenum EQ lein-lenum.

  IF sy-subrc NE 0.
    MESSAGE 'La UA introducida no existe' TYPE 'E'.
  ELSE.
    SELECT SINGLE * FROM makt
      WHERE spras EQ sy-langu
        AND matnr EQ lqua-matnr.

    DATA(out_meins) = COND meins( WHEN lqua-letyp EQ 'BB' THEN 'M'
                                  WHEN lqua-letyp EQ 'PL' THEN 'ST' ).

    CALL FUNCTION 'MD_CONVERT_MATERIAL_UNIT'
      EXPORTING
        i_matnr              = lqua-matnr
        i_in_me              = lqua-meins
        i_out_me             = out_meins
        i_menge              = lqua-verme
      IMPORTING
        e_menge              = lqua-verme
      EXCEPTIONS
        error_in_application = 1
        error                = 2
        OTHERS               = 3.

    lqua-verme = trunc( lqua-verme ).
    lqua-meins = out_meins.
  ENDIF.
ENDMODULE.
