        IF sy-subrc = 0.

            f_flag = 'X'.
          ELSE. "Si el flag ya est relleno, no es la primera columna, concatenar la columna a lo que ya estaba en la cabecera y separarlo con gc_tab
            CONCATENATE lv_cabecera gc_tab wa_desc-scrtext_s INTO lv_cabecera.
          ENDIF.
        ENDIF.

      ENDIF.
    ENDLOOP.

        EXPORTING
