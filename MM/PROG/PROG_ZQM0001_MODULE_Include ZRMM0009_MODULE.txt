*&---------------------------------------------------------------------*

*&---------------------------------------------------------------------*
*&      Module  STATUS_0100  OUTPUT
*&---------------------------------------------------------------------*
*       Status, título y llamada a ALV
*----------------------------------------------------------------------*
MODULE status_0100 OUTPUT.

  SET PF-STATUS 'MAIN'.
  SET TITLEBAR  'MAIN'.

  PERFORM alv_mostrar.

ENDMODULE.                 " STATUS_0100  OUTPUT

*&---------------------------------------------------------------------*
*&      Module  USER_COMMAND_0100  INPUT
*&---------------------------------------------------------------------*
*       Acciones USER COMMAND 0100
*----------------------------------------------------------------------*
MODULE user_command_0100 INPUT.

  CASE gv_okcode.

*   Salir
    WHEN 'EXIT' OR 'CANC'.

      SET SCREEN 0.
      LEAVE SCREEN.

  ENDCASE.

  CLEAR: gv_okcode.

ENDMODULE.                 " USER_COMMAND_0100  INPUT
