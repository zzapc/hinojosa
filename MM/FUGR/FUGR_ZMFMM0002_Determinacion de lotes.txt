FUNCTION zmfmm0002.
*"----------------------------------------------------------------------
*"*"Interfase local
*"  IMPORTING
*"     REFERENCE(I_WERKS) TYPE  WERKS_D OPTIONAL
*"     REFERENCE(I_MTART) TYPE  MTART OPTIONAL
*"     REFERENCE(I_MATNR) TYPE  MATNR OPTIONAL
*"  EXPORTING
*"     REFERENCE(O_CHARG) TYPE  CHARG_D
*"  EXCEPTIONS
*"      CENTRO_NO_CONTEMPLADO
*"      NO_SUJETO_LOTE
*"----------------------------------------------------------------------


  DATA: lv_week TYPE kweek.
  DATA: lv_zyear    TYPE zed_year,
        lv_zweek    TYPE zed_week,
        lv_zempresa TYPE zed_empresa,
        lv_mtart    TYPE mtart,
        lv_ztipomat TYPE zed_tipomat.

* EDIEGO 32MM003 02/03/2016 -->
  FIELD-SYMBOLS: <fs_mb_xfeld> TYPE c.
  DATA: lv_str(80).
* EDIEGO 32MM003 02/03/2016 <--

  CLEAR: gv_functioncall1, gv_numero_global, lv_week, lv_zyear, lv_zweek, lv_zempresa, lv_ztipomat.

  SELECT SINGLE * " cvivo - 06.03.2019 - 50651 verificación a nivel de centro
  FROM marc
  WHERE matnr = i_matnr
    AND werks = i_werks
    AND xchpf = 'X'.

  IF sy-subrc = 0.
*   TPALLARES 32LO002 03/05/2017 -->
*   EDIEGO 32MM003 02/03/2016 -->
*   Para el centro 4000 “Cipasi” = Lote (10 dígitos) = CP99999999
*    IF i_werks EQ '4000'.
*      o_charg = 'CP99999999'.
*
**   Limpiamos variable CHECK_BATCH
*    lv_str = '(SAPLV01Z)CALL_VB_CREATE'.
*    ASSIGN (lv_str) TO <fs_MB_XFELD>.
*    IF sy-subrc = 0.
*      <fs_MB_XFELD> = 'X'.
*    ENDIF.
*
*    UNASSIGN <fs_MB_XFELD>.
*      EXIT.
*    ENDIF.
*   EDIEGO 32MM003 02/03/2016 <--
*   TPALLARES 32LO002 03/05/2017 <--

    IF i_mtart IS INITIAL.
      lv_mtart = mara-mtart.
    ELSE.
      lv_mtart = i_mtart.
    ENDIF.


* RECUPERAMOS EL AÑO Y SEMANA DEL AÑO
    CALL FUNCTION 'DATE_GET_WEEK'
      EXPORTING
        date         = sy-datum
      IMPORTING
        week         = lv_week
      EXCEPTIONS
        date_invalid = 1
        OTHERS       = 2.

    lv_zyear = lv_week+3(1).
    lv_zweek = lv_week+4(2).

* RECUPERAMOS EL CODIGO DE LA EMPRESA
    CASE i_werks.
      WHEN '2000'.
        lv_zempresa = 'X'.
      WHEN '2001' OR '2003'.
        lv_zempresa = 'H'.
      WHEN '2004'.
        lv_zempresa = 'F'.
      WHEN '2010'.
        lv_zempresa = 'R'.
      WHEN '2020'.
        lv_zempresa = 'O'.
      WHEN '2030'.
        lv_zempresa = 'V'.
      WHEN '2060'.
        lv_zempresa = 'G'.
      WHEN '3000'.
        lv_zempresa = 'P'.
      WHEN '3020'.
        lv_zempresa = 'A'.
*   TPALLARES 32LO002 03/05/2017 -->
      WHEN '4000'.
        lv_zempresa = 'C'.
*   TPALLARES 32LO002 03/05/2017 <--
      WHEN '2070'.
        lv_zempresa = 'S'.
* AMA 06/03/2019 se añade CB
      WHEN '2040' OR '2041'.
        lv_zempresa = 'B'.
      WHEN '2051' OR '2052'. " Demco - cvivo - 02.10.2019
        lv_zempresa = 'D'.
      WHEN OTHERS.
        RAISE centro_no_contemplado.
        EXIT.
    ENDCASE.

* RECUPERAMOS EL CODIGO DEL TIPO DE MATERIAL
    CASE lv_mtart.
      WHEN 'ZPAP'.
        lv_ztipomat = 'B'.
      WHEN 'ZCAR'.
        lv_ztipomat = 'C'.
      WHEN 'ZMAP'.
        lv_ztipomat = 'M'.
      WHEN 'ZAUX'.
        lv_ztipomat = 'A'.
      WHEN 'ZREC'.
        lv_ztipomat = 'R'.
      WHEN 'ZROL'.
        lv_ztipomat = 'S'.
      WHEN 'ZCAH'.
        lv_ztipomat = 'H'.
*      WHEN 'ZPLA'.
*        lv_ztipomat = 'P'.
      WHEN 'ZCAB'.
        lv_ztipomat = 'C'.
*   TPALLARES 32LO002 03/05/2017 -->
      WHEN 'ZPLA'.
        lv_ztipomat = 'E'.
      WHEN 'ZPLP'.
        lv_ztipomat = 'P'.
      WHEN 'ZPMP'.
        lv_ztipomat = 'T'.
      WHEN 'ZWIP'.
        lv_ztipomat = 'W'.
*   TPALLARES 32LO002 03/05/2017 <--
      WHEN OTHERS.
        lv_ztipomat = 'O'.
    ENDCASE.


    CALL FUNCTION 'ZMFMM0003'
      STARTING NEW TASK 'FUNC1' DESTINATION 'NONE'
      PERFORMING set_function1_done ON END OF TASK
      EXPORTING
        i_zyear    = lv_zyear
        i_zweek    = lv_zweek
        i_zempresa = lv_zempresa
        i_ztipomat = lv_ztipomat.
    WAIT UNTIL gv_functioncall1 = c_marca.

    CONCATENATE lv_zyear lv_zweek lv_zempresa lv_ztipomat gv_numero_global INTO o_charg.

  ELSE.

  ENDIF.


ENDFUNCTION.
