FUNCTION zpp_dimensiones.
*"----------------------------------------------------------------------
*"*"Interfase local
*"  IMPORTING
*"     VALUE(I_WERKS) TYPE  WERKS_D OPTIONAL
*"     VALUE(I_MATKL) TYPE  MATKL OPTIONAL
*"     VALUE(I_MATNR) TYPE  MATNR OPTIONAL
*"     VALUE(I_KUNNR) TYPE  KUNNR OPTIONAL
*"  EXPORTING
*"     VALUE(E_DIMENSIONES) TYPE  ZTPP0011
*"----------------------------------------------------------------------

* Para la lectura de los registros de esta tabla, se creará una secuencia
* de acceso siguiendo el siguiente orden:
*  1.	Centro, Material y Cliente
*  2.	Centro, Cliente
*  3.	Centro, Material
*  4.	Centro, Grupo de artículos
*  5.	Centro


  SELECT SINGLE * FROM ZTPP0011 "1.Centro, Material y Cliente
    INTO e_dimensiones
    WHERE werks = i_werks AND
          matnr = i_matnr AND
          kunnr = i_kunnr.
  IF sy-subrc <> 0.
    SELECT SINGLE * FROM ZTPP0011 "2.Centro, Cliente
      INTO e_dimensiones
      WHERE werks = i_werks AND
            kunnr = i_kunnr.
    IF sy-subrc <> 0.
      SELECT SINGLE * FROM ZTPP0011 "3.Centro, Material
        INTO e_dimensiones
        WHERE werks = i_werks AND
              matnr = i_matnr.
      IF sy-subrc <> 0.
        SELECT SINGLE * FROM ZTPP0011 "4.Centro, Grupo de artículos
          INTO e_dimensiones
          WHERE werks = i_werks AND
                matkl = i_matkl.
        IF sy-subrc <> 0.
          SELECT SINGLE * FROM ZTPP0011 "5.Centro
            INTO e_dimensiones
            WHERE werks = i_werks.
        ENDIF.
      ENDIF.
    ENDIF.
  ENDIF.



ENDFUNCTION.
