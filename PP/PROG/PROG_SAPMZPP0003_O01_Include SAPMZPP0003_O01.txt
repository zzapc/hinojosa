MODULE status_0100 OUTPUT.

  DATA: lt_codes TYPE TABLE OF sy-ucomm.

  REFRESH lt_codes.
  IF gv_log_handle IS INITIAL.
    APPEND 'LOG' TO lt_codes.
  ENDIF.

  SET PF-STATUS 'STAT100' EXCLUDING lt_codes.
  SET TITLEBAR 'TIT100'.
ENDMODULE.
