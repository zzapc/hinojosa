*----------------------------------------------------------------------*
MODULE status OUTPUT.
  SET PF-STATUS '0100'.

  IF gr_alv IS NOT BOUND.
    PERFORM crear_alv.
  ENDIF.

ENDMODULE.
*&---------------------------------------------------------------------*
*&      Module  USER_COMMAND_0100  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE user_command_0100 INPUT.

  CASE sy-ucomm.
    WHEN 'EXIT' OR 'BACK' OR 'CANCEL'.
      SET SCREEN 0.

  ENDCASE.
ENDMODULE.
