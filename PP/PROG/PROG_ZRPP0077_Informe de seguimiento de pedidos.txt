REPORT zrpp0077.

INCLUDE zrpp0077_top.
INCLUDE zrpp0077_class.
INCLUDE zrpp0077_forms.
INCLUDE zrpp0077_status_0100o01.

INITIALIZATION.
  SELECT SINGLE parva FROM usr05
    INTO p_werks
    WHERE bname = sy-uname AND
          parid = 'WRK'.

  p_comp = 'X'.
  p_vent = ' '.

AT SELECTION-SCREEN ON VALUE-REQUEST FOR p_lay06.
  PERFORM buscar_variante CHANGING p_lay06.

AT SELECTION-SCREEN ON VALUE-REQUEST FOR s_zsegca-low.

  SELECT  zzsegal , texto
    INTO TABLE @DATA(lt_zzsegcal)
    FROM ztsd0004
    WHERE spras = @sy-langu
    ORDER BY zzsegal.

*Buscamos solo los que sean del país Perú (PE)

*Se hace uso de la función que convoca al matchcode y transforma nuestra tabla interna
  CALL FUNCTION 'F4IF_INT_TABLE_VALUE_REQUEST'
    EXPORTING
      retfield        = 'ZZSEGAL'
      window_title    = 'Código seg.calidad'
      value_org       = 'S'
    TABLES
      value_tab       = lt_zzsegcal
      return_tab      = it_match
    EXCEPTIONS
      parameter_error = 1
      no_values_found = 2
      OTHERS          = 3.

  IF sy-subrc = 0.
    TRY.
        s_zsegca = VALUE #( sign = 'I' option = 'EQ' low = it_match[ 1 ]-fieldval ).
      CATCH cx_root .
        CLEAR s_zsegca.
    ENDTRY.
  ENDIF.

AT SELECTION-SCREEN ON VALUE-REQUEST FOR s_zsegca-high.

  SELECT  zzsegal , texto
    INTO TABLE @DATA(lt_zzsegcal)
    FROM ztsd0004
    WHERE spras = @sy-langu
    ORDER BY zzsegal.

*Buscamos solo los que sean del país Perú (PE)

*Se hace uso de la función que convoca al matchcode y transforma nuestra tabla interna
  CALL FUNCTION 'F4IF_INT_TABLE_VALUE_REQUEST'
    EXPORTING
      retfield        = 'ZZSEGAL'
      window_title    = 'Código seg.calidad'
      value_org       = 'S'
    TABLES
      value_tab       = lt_zzsegcal
      return_tab      = it_match
    EXCEPTIONS
      parameter_error = 1
      no_values_found = 2
      OTHERS          = 3.

  IF sy-subrc = 0.
    TRY.
        s_zsegca = VALUE #( sign = 'I' option = 'EQ' high = it_match[ 1 ]-fieldval ).
      CATCH cx_root .
        CLEAR s_zsegca.
    ENDTRY.
  ENDIF.

AT SELECTION-SCREEN.
  IF p_comp = ' ' AND
     p_vent = ' ' .
    MESSAGE |Seleccionar compras o ventas.| TYPE 'E' DISPLAY LIKE 'I'.
    RETURN.
  ENDIF.

  IF s_lfdat[] IS  INITIAL AND
     s_bedat   IS  INITIAL AND
     s_vbeln   IS INITIAL.
    MESSAGE |Introducir una fecha de documento, una fecha de entrega o un número de pedido.|
            TYPE 'E' DISPLAY LIKE 'I'.
    RETURN.

  ENDIF.



START-OF-SELECTION.

  PERFORM obtener_datos tables gt_datos.

END-OF-SELECTION.
  IF gt_datos IS INITIAL.
    MESSAGE |No se han encontrado datos.| TYPE 'I' DISPLAY LIKE 'E'.
  ELSE.
*    call screen 100.
    PERFORM crear_alv.
  ENDIF.
