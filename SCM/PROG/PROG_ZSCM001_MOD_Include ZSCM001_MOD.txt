*&---------------------------------------------------------------------*
*&      Module  USER_COMMAND_9000  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE user_command_9000 INPUT.

  CASE sy-ucomm.
    WHEN '&F03' OR '&F15'.
      LEAVE TO SCREEN 0.

    WHEN '&F12'.
      LEAVE PROGRAM.

  ENDCASE.

ENDMODULE.

MODULE status_9100 OUTPUT.
*  SET PF-STATUS 'xxxxxxxx'.
*  SET TITLEBAR 'xxx'.
ENDMODULE.
*&---------------------------------------------------------------------*
*&      Module  USER_COMMAND_9100  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE user_command_9100 INPUT.
  CASE sy-ucomm.
    WHEN 'BT_OK'.
      IF gv_werks IS NOT INITIAL AND gv_lgort IS NOT INITIAL AND gv_kunnr IS NOT INITIAL.
        SET SCREEN 0.
        LEAVE SCREEN.
      ELSE.
        MESSAGE 'Debe informar los tres campos.' TYPE 'I'.
      ENDIF.

    WHEN 'BT_KO'.
      SET SCREEN 0.
      LEAVE SCREEN.

  ENDCASE.
ENDMODULE.
*&---------------------------------------------------------------------*

MODULE status_9200 OUTPUT.

*  SET TITLEBAR 'xxx'.

*&---------------------------------------------------------------------*
*&      Module  USER_COMMAND_9200  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE user_command_9200 INPUT.
  CASE sy-ucomm.
    WHEN 'BT_OK'.
      IF gv_clabs IS NOT INITIAL.
        SET SCREEN 0.
        LEAVE SCREEN.
      ELSE.
        MESSAGE 'Debe informar la cantidad de la posición.' TYPE 'I'.
      ENDIF.

    WHEN 'BT_KO'.
      CLEAR gv_clabs.
      SET SCREEN 0.
      LEAVE SCREEN.

  ENDCASE.
ENDMODULE.
