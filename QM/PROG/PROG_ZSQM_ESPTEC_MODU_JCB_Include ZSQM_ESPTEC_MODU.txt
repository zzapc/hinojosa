*&---------------------------------------------------------------------*
*&  Include           ZSQM_ESPTEC_MODU
*&---------------------------------------------------------------------*
*&---------------------------------------------------------------------*
*& Module STATUS_2000 OUTPUT
*&---------------------------------------------------------------------*
MODULE status_2000 OUTPUT.

  SET PF-STATUS '2000'.
  SET TITLEBAR  '2000'.

  variant-report = sy-repid.
  variant-handle = '0001'.
  PERFORM load_data_into_grid TABLES gt_output
                              USING   'ALV' variant 'ZSTQM_ESPTEC'
                              CHANGING container alv layout fieldcat excludin.
ENDMODULE.

*&---------------------------------------------------------------------*
*&      Module  USER_COMMAND_2000  INPUT
*&---------------------------------------------------------------------*
MODULE user_command_2000 INPUT.

  DATA lv_answer TYPE char1.

  CASE sy-ucomm.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'EXIT'.
      LEAVE PROGRAM.
    WHEN 'REFRESH'.
      PERFORM buscar_datos.
    WHEN 'FORM'.
      CALL METHOD alv->get_selected_rows
        IMPORTING
          et_index_rows = lt_rows.

*        MESSAGE s368(00) WITH 'Solo es posible seleccionar una línea'.
      IF lt_rows[] IS INITIAL.
        MESSAGE s368(00) WITH 'Es necesario seleccionar una línea para obtener el formulario'.
      ELSE.
        IF lines( lt_rows ) GT 1.
          CALL FUNCTION 'POPUP_TO_CONFIRM'
            EXPORTING
              text_question         = 'Has seleccionado varias entradas. ¿Deseas continuar?'
              display_cancel_button = ' '
            IMPORTING
              answer                = lv_answer
            EXCEPTIONS
              text_not_found        = 1
              OTHERS                = 2.
        ELSE.
          lv_answer = 1.
        ENDIF.
        IF lv_answer = 1.
          LOOP AT lt_rows INTO ls_rows.
            READ TABLE gt_output INTO ls_output INDEX ls_rows-index.
            PERFORM generar_form.
          ENDLOOP.
        ENDIF.
      ENDIF.

    WHEN 'DESCARGAR'.
      CALL METHOD alv->get_selected_rows
        IMPORTING
          et_index_rows = lt_rows.

*        MESSAGE s368(00) WITH 'Solo es posible seleccionar una línea'.
      IF lt_rows[] IS INITIAL.
        MESSAGE s368(00) WITH 'Es necesario seleccionar una línea para obtener el formulario'.
      ELSE.


        DATA : lv_path     TYPE string,
               lv_pathname TYPE string.

        CALL METHOD cl_gui_frontend_services=>directory_browse
          CHANGING
            selected_folder      = lv_path
          EXCEPTIONS
            cntl_error           = 1
            error_no_gui         = 2
            not_supported_by_gui = 3
            OTHERS               = 4.

        IF sy-subrc EQ 0.
          lv_pathname = lv_path.

          LOOP AT lt_rows INTO ls_rows.
            READ TABLE gt_output INTO ls_output INDEX ls_rows-index.
            PERFORM descargar_form USING lv_path.
          ENDLOOP.
        ENDIF.
      ENDIF.

  ENDCASE.

ENDMODULE.
