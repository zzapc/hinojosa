FUNCTION dmee_exit_csb68_num_pago.
*"----------------------------------------------------------------------
*"*"Interfase local
*"  IMPORTING
*"     VALUE(I_TREE_TYPE) TYPE  DMEE_TREETYPE
*"     VALUE(I_TREE_ID) TYPE  DMEE_TREEID
*"     VALUE(I_ITEM)
*"     VALUE(I_PARAM)
*"     VALUE(I_UPARAM)
*"  EXPORTING
*"     REFERENCE(O_VALUE)
*"     REFERENCE(C_VALUE)
*"     REFERENCE(N_VALUE)
*"     REFERENCE(P_VALUE)
*"  TABLES
*"      I_TAB
*"----------------------------------------------------------------------

* Template function module --------------------------------------------*

  DATA: l_item  TYPE dmee_paym_if_type,
        l_fpayh TYPE fpayh,
        l_doc1r TYPE fpayh-doc1r,
        i_numc  TYPE n LENGTH 11,
        anyo(2) TYPE n,
        doc(5)  TYPE n.

  l_item = i_item.
  l_fpayh = l_item-fpayh.

* Dígito de control número de pago CSB68
* Módulo de cálculo 7 (resto división por 7)
* Número 90 (pago residentes) + 00 (uso interno emisor pagos) + número pago (DOC1R).

  anyo    = l_fpayh-doc1r+16(4) + 10.
  doc     = l_fpayh-doc1r+9(5).

  CONCATENATE anyo doc INTO c_value.

  CLEAR i_numc.
  i_numc = c_value.
  i_numc = 90000000000 + i_numc.
  i_numc = i_numc MOD 7.

  CONCATENATE c_value i_numc+10(1) INTO c_value.



ENDFUNCTION.
