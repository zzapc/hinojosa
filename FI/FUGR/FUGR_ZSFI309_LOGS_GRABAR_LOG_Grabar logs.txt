FUNCTION ZSFI309_LOGS_GRABAR_LOG.
*"----------------------------------------------------------------------
*"*"Interfase local
*"  IMPORTING
*"     REFERENCE(IV_LOGHANDLE) TYPE  BALLOGHNDL
*"----------------------------------------------------------------------

 DATA lt_loghandle TYPE bal_t_logh.

  CLEAR: lt_loghandle[].
  APPEND iv_loghandle TO lt_loghandle.

  CALL FUNCTION 'BAL_DB_SAVE'
    EXPORTING
      i_save_all       = 'X'
      i_t_log_handle   = lt_loghandle
    EXCEPTIONS
      log_not_found    = 1
      save_not_allowed = 2
      numbering_error  = 3
      OTHERS           = 4.
  IF sy-subrc <> 0.
    WRITE:/ 'ERROR guardando LOGs en BD. Código ', sy-subrc.
  ENDIF.

ENDFUNCTION.
