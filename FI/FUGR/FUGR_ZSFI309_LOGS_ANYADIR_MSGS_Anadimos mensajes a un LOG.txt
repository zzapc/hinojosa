FUNCTION zsfi309_logs_anyadir_msgs.
*"----------------------------------------------------------------------
*"*"Interfase local
*"  IMPORTING
*"     REFERENCE(IV_LOGHANDLE) TYPE  BALLOGHNDL
*"     REFERENCE(IV_TABLA) TYPE  FLAG OPTIONAL
*"     REFERENCE(IV_MSG) TYPE  BAL_S_MSG OPTIONAL
*"  TABLES
*"      T_MSGS TYPE  BAL_T_MSG OPTIONAL
*"----------------------------------------------------------------------
  DATA ls_msg TYPE bal_s_msg.

  IF iv_tabla = ''.

    ls_msg = iv_msg.
    CALL FUNCTION 'BAL_LOG_MSG_ADD'
      EXPORTING
        i_log_handle     = iv_loghandle
        i_s_msg          = ls_msg
      EXCEPTIONS
        log_not_found    = 1
        msg_inconsistent = 2
        log_is_full      = 3
        OTHERS           = 4.
    IF sy-subrc <> 0.
      WRITE:/ 'ERROR registrando LOG. Código ', sy-subrc.
    ENDIF.

  ELSEIF iv_tabla = 'X'.

    LOOP AT t_msgs INTO ls_msg.

      CALL FUNCTION 'BAL_LOG_MSG_ADD'
        EXPORTING
          i_log_handle     = iv_loghandle
          i_s_msg          = ls_msg
        EXCEPTIONS
          log_not_found    = 1
          msg_inconsistent = 2
          log_is_full      = 3
          OTHERS           = 4.
      IF sy-subrc <> 0.
        WRITE:/ 'ERROR registrando LOG. Código ', sy-subrc.
      ENDIF.

    ENDLOOP.

  ENDIF.

ENDFUNCTION.
