MODULE dynpro_cyc OUTPUT.

  DATA: l_kkber TYPE kkber,
        vl_tipo TYPE zssdt_cyc_custo-tipo.


  SELECT SINGLE kkber INTO l_kkber
    FROM zssd_cyc_soc
    WHERE kkber = rf02l-kkber.
  IF sy-subrc = 0.

    " Inicio BG 21.11.2023 - Se busca tipo de operación
    SELECT SINGLE tipo FROM zssdt_cyc_custo
      INTO vl_tipo
      WHERE kkber = rf02l-kkber.
    " Fin BG 21.11.2023 - Se busca tipo de operación

    LOOP AT SCREEN.
      IF screen-group1 = 'CYC'.
        screen-invisible   = 0.
        MODIFY SCREEN.
      ENDIF.

      " Inicio BG 08.11.2023 - Si está en la tabla zssd_cyc_soc Bloquea el campo limite de credito para input
      IF vl_tipo NE '001'.
        IF screen-name = 'KNKK-KLIMK'.
          screen-input     = 0.
          MODIFY SCREEN.
        ENDIF.
      ENDIF.
      " Fin BG 08.11.2023 - Si está en la tabla zssd_cyc_soc Bloquea el campo limite de credito para input
    ENDLOOP.
  ELSE.
    LOOP AT SCREEN.
      IF screen-group1 = 'CYC'.
        screen-invisible   = 1.
        " Inicio BG 08.11.2023 - Nuevos campos Riesgo Interno GAPSD048
        IF screen-name = 'KNKK-ZZRIESGO'.
          screen-input     = 0.
        ENDIF.
        " Fin BG 08.11.2023 - Nuevos campos Riesgo Interno GAPSD048
        MODIFY SCREEN.
      ENDIF.
    ENDLOOP.
  ENDIF.

  " Inicio BG 08.11.2023 - Nuevos campos Riesgo Interno GAPSD048
  IF sy-tcode = 'FD33'.
    LOOP AT SCREEN.
      IF screen-name = 'KNKK-ZZRIESGO'.
        screen-input   = 0.
        MODIFY SCREEN.
      ENDIF.
    ENDLOOP.
  ENDIF.
  " Fin BG 08.11.2023 - Nuevos campos Riesgo Interno GAPSD048
ENDMODULE.
