FUNCTION zssdfm023.
*"----------------------------------------------------------------------
*"*"Interfase local
*"  IMPORTING
*"     REFERENCE(OFERTA) TYPE  VBELN_VA
*"     REFERENCE(PEDIDO) TYPE  VBELN_VA
*"     REFERENCE(TCODE) TYPE  SY-TCODE OPTIONAL
*"  TABLES
*"      XVBAP STRUCTURE  VBAPVB
*"----------------------------------------------------------------------

  DATA: name       TYPE thead-tdname,
        header     TYPE thead,
        lt_lines   TYPE STANDARD TABLE OF tline,
        lt_lines2  TYPE STANDARD TABLE OF tline,
        lt_pos_aux TYPE STANDARD TABLE OF vbapvb.

  IF tcode = 'VA22'.

    lt_pos_aux = xvbap[].
    DELETE lt_pos_aux WHERE matnr = 'SINGLE'.

    " Agrega texto de posición generica SINGlE a posición definitiva
    LOOP AT xvbap INTO DATA(ls_pos_oferta) WHERE matnr = 'SINGLE'.
      CLEAR: name, header, lt_lines[], lt_lines2[].
      READ TABLE lt_pos_aux INTO DATA(ls_pos_aux) WITH KEY posnr+3(3) = ls_pos_oferta-posnr+3(3).
      IF sy-subrc EQ 0.
        " lee ZC04
        name = ls_pos_oferta-vbeln && ls_pos_oferta-posnr.

        CALL FUNCTION 'READ_TEXT'
          EXPORTING
            id                      = 'ZC04'
            language                = sy-langu
            name                    = name
            object                  = 'VBBP'
          TABLES
            lines                   = lt_lines
          EXCEPTIONS
            id                      = 1
            language                = 2
            name                    = 3
            not_found               = 4
            object                  = 5
            reference_check         = 6
            wrong_access_to_archive = 7
            OTHERS                  = 8.
        IF sy-subrc <> 0.
* Implement suitable error handling here
          CONTINUE.
        ENDIF.

        IF lt_lines[] IS NOT INITIAL.
          " Revisa si la posición definitiva no tiene texto
          CLEAR name.
          name = ls_pos_aux-vbeln && ls_pos_aux-posnr.

          CALL FUNCTION 'READ_TEXT'
            EXPORTING
              id                      = 'ZC04'
              language                = sy-langu
              name                    = name
              object                  = 'VBBP'
            TABLES
              lines                   = lt_lines2
            EXCEPTIONS
              id                      = 1
              language                = 2
              name                    = 3
              not_found               = 4
              object                  = 5
              reference_check         = 6
              wrong_access_to_archive = 7
              OTHERS                  = 8.
          IF sy-subrc <> 0.
* Implement suitable error handling here
          ENDIF.

          IF lt_lines[] NE lt_lines2[].
            header-tdobject = 'VBBP'.
            header-tdname = name.
            header-tdid = 'ZC04'.
            header-tdspras = sy-langu.

            CALL FUNCTION 'SAVE_TEXT'
              EXPORTING
                header          = header
                savemode_direct = abap_true
              TABLES
                lines           = lt_lines
              EXCEPTIONS
                id              = 1
                language        = 2
                name            = 3
                object          = 4
                OTHERS          = 5.
            IF sy-subrc <> 0.
* Implement suitable error handling here
            ENDIF.
          ENDIF.
        ENDIF.
      ENDIF.
      CLEAR: ls_pos_oferta,  ls_pos_aux.
    ENDLOOP.
  ENDIF.

  IF tcode = 'VA01'.
    CLEAR ls_pos_oferta.
    LOOP AT xvbap INTO ls_pos_oferta WHERE vgbel = oferta.
      " lee ZC04
      name = ls_pos_oferta-vgbel && ls_pos_oferta-vgpos.

      CALL FUNCTION 'READ_TEXT'
        EXPORTING
          id                      = 'ZC04'
          language                = sy-langu
          name                    = name
          object                  = 'VBBP'
        TABLES
          lines                   = lt_lines
        EXCEPTIONS
          id                      = 1
          language                = 2
          name                    = 3
          not_found               = 4
          object                  = 5
          reference_check         = 6
          wrong_access_to_archive = 7
          OTHERS                  = 8.
      IF sy-subrc <> 0.
* Implement suitable error handling here
      ENDIF.

      " Pasa el texto de ZC04 a ZC05
      IF sy-subrc EQ 0.
        header-tdobject = 'VBBP'.
        header-tdname = name.
        header-tdid = 'ZC05'.
        header-tdspras = sy-langu.

        CALL FUNCTION 'SAVE_TEXT'
          EXPORTING
            header          = header
            savemode_direct = abap_true
          TABLES
            lines           = lt_lines
          EXCEPTIONS
            id              = 1
            language        = 2
            name            = 3
            object          = 4
            OTHERS          = 5.
        IF sy-subrc <> 0.
* Implement suitable error handling here
        ENDIF.

        " Borra ZC04
        IF sy-subrc EQ 0.
          CALL FUNCTION 'DELETE_TEXT'
            EXPORTING
              id              = 'ZC04'
              language        = sy-langu
              name            = name
              object          = 'VBBP'
              savemode_direct = abap_true
            EXCEPTIONS
              not_found       = 1
              OTHERS          = 2.
          IF sy-subrc <> 0.
* Implement suitable error handling here
          ENDIF.
        ENDIF.
      ENDIF.

      CLEAR: ls_pos_oferta, name, header, lt_lines[].
    ENDLOOP.
  ENDIF.

ENDFUNCTION.
